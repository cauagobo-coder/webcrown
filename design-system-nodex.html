CSS Tokens & Config

Copie estes tokens para o seu style.css ou na config do Tailwind tailwind.config.js.

/* ── WEB CROWN Design Tokens v2 ── */
:root {
  /* Paleta */
  --color-gold:        #C4A93C;
  --color-light-gray:  #DCDCDC;
  --color-black:       #0A0A0A;
  --color-sky:         #C8DCF0;

  /* Backgrounds */
  --bg-base:           #0A0A0A;
  --bg-surface-1:      #111111;
  --bg-surface-2:      #1A1A1A;
  --bg-surface-3:      #242424;

  /* Textos */
  --text-main:         #FFFFFF;
  --text-secondary:    #DCDCDC;
  --text-muted:        #888888;

  /* Accents */
  --accent-gold:       #C4A93C;
  --accent-sky:        #C8DCF0;

  /* Bordas */
  --border-color:      rgba(255, 255, 255, 0.08);

  /* Fontes */
  --font-display:      'ClashDisplay', 'Inter', sans-serif;
  --font-body:         'Inter', sans-serif;

  /* Escala Tipográfica */
  --type-h1:   96px;  /* Clash Display Semibold  */
  --type-h2:   48px;  /* Clash Display Medium     */
  --type-h3:   34px;  /* Clash Display Medium     */
  --type-h4:   20px;  /* Clash Display Medium     */
  --type-sub1: 16px;  /* Clash Display Medium     */
  --type-sub2: 16px;  /* Clash Display Regular    */
  --type-body: 16px;  /* Inter Regular            */

  /* Radius */
  --radius-sm:   4px;
  --radius-base: 8px;
  --radius-lg:   16px;
  --radius-xl:   24px;
  --radius-full: 9999px;
}

/* Google / Fontshare Imports */
@import url('https://api.fontshare.com/v2/css?f[]=clash-display@200,300,400,500,600,700&display=swap');
@import url('https://fonts.googleapis.com/css2?family=Inter:wght@300;400;500;600;700&display=swap');

/* Tailwind Custom Extended Config */
module.exports = {
  theme: {
    extend: {
      fontFamily: {
        display: ['ClashDisplay', 'Inter', 'sans-serif'],
        body:    ['Inter', 'sans-serif'],
      },
      colors: {
        brand: {
          gold:  '#C4A93C',
          gray:  '#DCDCDC',
          black: '#0A0A0A',
          sky:   '#C8DCF0',
        }
      }
    }
  }
}
